109float debug_servo_run(int32_t dt, PtpServoAuxInput *pAux) {
110 double tuning_ppb = 0.0;
111
112 // in the very first cycle skip running the filter
113 if (cycle == 0) {
114 goto retain_cycle_data;
115 }
116
117 // calculate input data
118 double skew = ((double)(dt - dt_prev)) / ((double)pAux->measSyncPeriodNs) * 1E+09; // skew
119 double skew_rel = skew - skew0; // substract skew offset
120 int32_t offset = dt - dt0 - skew_rel * pAux->measSyncPeriodNs * 1E-09; // time offset
121 int32_t offset_delta = offset - offset_prev;
122
123 MSG(PTP_COLOR_BGREEN "% 8.4f" PTP_COLOR_BYELLOW " % 12i" PTP_COLOR_BRED " % 12i" PTP_COLOR_RESET "\n", skew_rel, offset, offset_delta);
124
125 offset_prev = offset;
126 skew_prev = skew;
127
128retain_cycle_data:

130

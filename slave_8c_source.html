312void ptp_slave_process_message(RawPtpMessage *pRawMsg, PtpHeader *pHeader) {
313 PtpMessageType mt = pHeader->messageType;
314 PtpDelayMechanism dm = S.profile.delayMechanism;
315
316 // process non-Announce messages
317 if (mt == PTP_MT_Sync || mt == PTP_MT_Follow_Up) {
318 switch (S.slave.messaging.m2sState) {
319 // wait for Sync message
320 case SIdle: {
321 // switch into next state if Sync packet has arrived
322 if (mt == PTP_MT_Sync) {
323 // save sync interval
324 S.slave.messaging.logSyncPeriod = pHeader->logMessagePeriod;
325 S.slave.messaging.syncPeriodMs = ptp_logi2ms(pHeader->logMessagePeriod);
326
327 // MSG("%d\n", header.logMessagePeriod);
328
329 // save reception time
330 S.slave.scd.t[T2] = pRawMsg->ts;
331
332 // switch to next syncState
333 S.slave.messaging.sequenceID = pHeader->sequenceID;
334
335 // save correction field
336 S.slave.scd.cf[T1] = pHeader->correction_ns;
337
338 // handle two step/one step messaging
339 if (pHeader->flags.PTP_TWO_STEP) {
340 S.slave.messaging.m2sState = SWaitFollowUp;
341 } else {
342 ptp_extract_timestamps(&S.slave.scd.t[T1], pRawMsg->data, 1); // extract t1
343 S.slave.scd.cf[T2] = 0; // clear Follow_Up correction field, since no Follow_Up is expected
344 ptp_commence_e2e_correction(); // commence executing the E2E correction
345 }
346
347 // dispatch SYNC_RECVED event
348 PTP_IUEV(PTP_UEV_SYNC_RECVED);
349 }
350
351 break;
352 }
353 // wait for Follow_Up message
354 case SWaitFollowUp:
355 if (mt == PTP_MT_Follow_Up) {
356 // check sequence ID if the response is ours
357 if (pHeader->sequenceID == S.slave.messaging.sequenceID) {
358 ptp_extract_timestamps(&S.slave.scd.t[T1], pRawMsg->data, 1); // read t1
359 S.slave.scd.cf[T2] = pHeader->correction_ns; // retain correction field
360
361 // initiate the correction
362 ptp_commence_e2e_correction();
363
364 // log correction field (if enabled)
365 CLILOG(S.logging.corr, "C [Follow_Up]: %09lu\n", pHeader->correction_ns);
366
367 // dispatch FOLLOW_UP_RECVED event
368 PTP_IUEV(PTP_UEV_FOLLOW_UP_RECVED);
369 }
370
371 // on ID mismatch, just skip the cycle, and expect a new Sync coming
372
373 // switch to next syncState
374 S.slave.messaging.m2sState = SIdle;
375 }
376
377 break;
378 }
379 }
380
381 // ------ (P)DELAY_RESPONSE PROCESSING --------
382
383 // wait for (P)Delay_Resp message
384 if (((mt == PTP_MT_Delay_Resp) && (dm == PTP_DM_E2E)) ||
385 (((mt == PTP_MT_PDelay_Resp) || (mt == PTP_MT_PDelay_Resp_Follow_Up)) && (dm == PTP_DM_P2P))) {
386 if (pHeader->sequenceID == S.slave.messaging.delay_reqSequenceID) { // read clock ID of requester
387 PtpDelay_RespIdentification delay_respID; // identification received in every Delay_Resp packet
388
389 if (mt == PTP_MT_Delay_Resp) { // Delay_Resp processing
390
391 ptp_read_delay_resp_id_data(&delay_respID, pRawMsg->data);
392
393 // if the response was sent to us as a response to our Delay_Req then continue processing
394 if (delay_respID.requestingSourceClockIdentity == S.hwoptions.clockIdentity &&
395 delay_respID.requestingSourcePortIdentity == PTP_PORT_ID) {
396
397 ptp_extract_timestamps(&S.slave.scd.t[T4], pRawMsg->data, 1); // store t4
398 S.slave.scd.cf[T4] = pHeader->correction_ns; // store correction field
399
400 // compute mean path delay
401 ptp_compute_mean_path_delay_e2e(S.slave.scd.t, S.slave.scd.cf, &S.network.meanPathDelay);
402
403 // store last response ID
404 S.slave.messaging.lastRespondedDelReqId = pHeader->sequenceID;
405
406 // perform correction if operating on syncmatched mode
407 if (S.profile.logDelayReqPeriod == PTP_LOGPER_SYNCMATCHED) {
408 ptp_perform_correction();
409 }
410
411 // dispatch DELAY_RESP_RECVED event
412 PTP_IUEV(PTP_UEV_DELAY_RESP_RECVED);
413
414 // log correction field (if enabled)
415 CLILOG(S.logging.corr, "C [Del_Resp]: %09lu\n", pHeader->correction_ns);
416 }
417
418 } else if (mt == PTP_MT_PDelay_Resp) { // PDelay_Resp processing
419 TimestampI *pT = &S.slave.scd.t[2]; // skip the first 2 timestamps
420 uint64_t *cf = &S.slave.scd.cf[2]; // skip the first 2 correction fields
421
422 pT[T4] = pRawMsg->ts; // save t4 (P2P)
423 cf[T2] = pHeader->correction_ns; // save correction field of the PDelay_Resp
424
425 // if the responder is a one-step clock, then...
426 if (!pHeader->flags.PTP_TWO_STEP) {
427 // no t2 and t3 will be involved with the calculations
428 pT[T3] = pT[T2] = zeroTs;
429 ptp_commence_p2p_correction(pHeader->sequenceID); // commence correction
430 } else {
431 ptp_extract_timestamps(&(pT[T2]), pRawMsg->data, 1); // retrieve t2 (P2P)
432 }
433
434 // dispatch PDELAY_RESP_RECVED event
435 PTP_IUEV(PTP_UEV_PDELAY_RESP_RECVED);
436
437 // log correction field (if enabled)
438 CLILOG(S.logging.corr, "C [PDel_Resp]: %09lu\n", pHeader->correction_ns);
439
440 } else if (mt == PTP_MT_PDelay_Resp_Follow_Up) { // PDelay_Resp_Follow_Up processing
441 ptp_read_delay_resp_id_data(&delay_respID, pRawMsg->data);
442
443 // if sent to us as a response to our Delay_Req then continue processing
444 if (delay_respID.requestingSourceClockIdentity == S.hwoptions.clockIdentity &&
445 delay_respID.requestingSourcePortIdentity == PTP_PORT_ID) {
446
447 TimestampI *pT = &S.slave.scd.t[2]; // skip the first 2 timestamps
448 uint64_t *cf = &S.slave.scd.cf[2]; // skip the first 2 correction fields
449
450 ptp_extract_timestamps(&(pT[T3]), pRawMsg->data, 1); // retrieve t3 (P2P)
451 cf[T3] = pHeader->correction_ns; // retain correction field from the PDelay_Resp_Follow_Up
452
453 // commence correction
454 ptp_commence_p2p_correction(pHeader->sequenceID);
455
456 // dispatch PDELAY_RESP_FOLLOW_UP_RECVED event
457 PTP_IUEV(PTP_UEV_PDELAY_RESP_FOLLOW_UP_RECVED);
458
459 // log correction field (if enabled)
460 CLILOG(S.logging.corr, "C [PDel_Resp_Follow_Up]: %09lu\n", pHeader->correction_ns);
461 }
462 }
463 }

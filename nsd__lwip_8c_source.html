29 if (TP == PTP_TP_IPv4) {
30 err_t (*igmp_fn)(const ip_addr_t *, const ip_addr_t *) = join ? igmp_joingroup : igmp_leavegroup; // join or leave

32 if (DM == PTP_DM_E2E) {

40void ptp_nsd_init(PtpTransportType tp, PtpDelayMechanism dm) {
41 // lock LWIP core
42 LOCK_TCPIP_CORE();
43
44 // leave current IGMP group if applicable
45 ptp_nsd_igmp_join_leave(false);
46
47 // first, close all open connection blocks (zero CBDs won't cause trouble)
48 if (PTP_L4_EVENT != NULL) {
49 udp_disconnect(PTP_L4_EVENT);
50 udp_remove(PTP_L4_EVENT);
51 PTP_L4_EVENT = NULL;
52 }
53 if (PTP_L4_GENERAL != NULL) {
54 udp_disconnect(PTP_L4_GENERAL);
55 udp_remove(PTP_L4_GENERAL);
56 PTP_L4_GENERAL = NULL;
57 }
58
59 // calling either parameter with -1 just closes connections
60 if ((tp == -1) || (dm == -1)) {
61 // message transmission and reception is turned off
62 TP = -1;
63 DM = -1;
64 return;
65 }
66
67 // open only the necessary ones
68 if (tp == PTP_TP_IPv4) {
69 // open event and general connections
70 ip_addr_t addr = (dm == PTP_DM_E2E) ? PTP_IGMP_PRIMARY : PTP_IGMP_PEER_DELAY;
71 PTP_L4_EVENT = udp_new();
72 udp_bind(PTP_L4_EVENT, &addr, PTP_PORT_EVENT);
73 udp_recv(PTP_L4_EVENT, ptp_receive_cb, NULL);
74
75 PTP_L4_GENERAL = udp_new();
76 udp_bind(PTP_L4_GENERAL, &addr, PTP_PORT_GENERAL);
77 udp_recv(PTP_L4_GENERAL, ptp_receive_cb, NULL);
78 }
79
80 // store configuration
81 TP = tp;
82 DM = dm;
83
84 // join new IGMP group
85 ptp_nsd_igmp_join_leave(true);
86
87 // unlock LWIP core
88 UNLOCK_TCPIP_CORE();
89}

118void ptp_nsd_transmit_msg(RawPtpMessage *pMsg) {
119 if (pMsg == NULL) {
120 MSG("NULL!!!\n");
121 return;
122 }
123
124 PtpMessageClass mc = pMsg->tx_mc;
125
126 // allocate buffer
127 struct pbuf *p = NULL;
128 p = pbuf_alloc((TP == PTP_TP_IPv4) ? PBUF_TRANSPORT : PBUF_LINK, pMsg->size, PBUF_RAM);
129
130 // fill buffer
131 memcpy(p->payload, pMsg->data, pMsg->size);
132
133 // set transmit callback
134 p->tag = pMsg;
135 p->tx_cb = ptp_transmit_cb;
136
137 // lock LWIP core
138 LOCK_TCPIP_CORE();
139
140 // narrow down by transport type
141 if (TP == PTP_TP_IPv4) {
142 struct udp_pcb *conn = (mc == PTP_MC_EVENT) ? PTP_L4_EVENT : PTP_L4_GENERAL; // select connection by message type
143 uint16_t port = (mc == PTP_MC_EVENT) ? PTP_PORT_EVENT : PTP_PORT_GENERAL; // select port by message class
144 ip_addr_t ipaddr = (DM == PTP_DM_E2E) ? PTP_IGMP_PRIMARY : PTP_IGMP_PEER_DELAY; // select destination IP-address by delmech.
145 udp_sendto(conn, p, &ipaddr, port); // send packet
146 } else if (TP == PTP_TP_802_3) {
147 const uint8_t *ethaddr = (DM == PTP_DM_E2E) ? PTP_ETHERNET_PRIMARY : PTP_ETHERNET_PEER_DELAY; // select destination address by delmech.
148 ethernet_output(netif_default, p, (struct eth_addr *)netif_default->hwaddr, (struct eth_addr *)ethaddr, ETHERTYPE_PTP);
149 }
150
151 // unlock LWIP core
152 UNLOCK_TCPIP_CORE();
153
154 pbuf_free(p); // release buffer
155}
